Biographical timelines from Wikipedia Infoboxes

Kevin Stadler

2017-02-24

library(Rwikitimeline)
## Loading required package: ggplot2
getwikibio("Hieronymus Bosch")
## birth death 
##  1450  1516
painters <- buildtimeline(c("Hieronymus Bosch", "Michelangelo", "Pieter Bruegel the Elder"))
linguists <- buildtimeline(c("Johann Wolfgang von Goethe", "William Jones (philologist)", 
    "Wilhelm von Humboldt", "Jacob Grimm", "Otto Jespersen", "Ferdinand de Saussure", 
    "Edward Sapir", "Leonard Bloomfield", "Ludwig Wittgenstein"))
print(linguists)
